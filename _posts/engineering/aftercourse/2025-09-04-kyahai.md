          // Keep original text for non-numeric values
          const finalValue = num.textContent;
          if (!/^\d+/.test(finalValue)) {
            return;
          }
        });
      }
    });
  }, observerOptions);
  
  // Observe stats section
  const statsSection = document.querySelector('.sp-agri-stats');
  if (statsSection) {
    observer.observe(statsSection);
  }
});
